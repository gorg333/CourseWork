<script language=javascript>
n=prompt("Введите число уравнений системы:"); //Ввод данных
const Source = [] // Исходный масив 
const Results = [] // Масив ответов
let i, j, k // Вспомагательные переменные

for(i = 0; i < n; i++) {  // Заполнение матрицы
    let Source[i] = []; 
    for(j = 0; j <= n; j++) {
        let i[j] = [];
        Source[i][j] = prompt("Введите член матрицы:"); //Ввод данных
    }
};

document.write("<table border = '3'>" )
for(i=0; i<n; ++i) {
for(j=0; j <= n; ++j) {
    if(j === 0) document.write("<tr>");
    document.write("<td>" + Source[i][j] + "</td>");
    if(j === n) document.write("</tr>");
    }
}
document.write("/<table>");
Operation(n);
Search_Results();

function Operation(item) { 
for(item=0;item<(n-1);item++) { //Цикл выполнения итерраций
if (Source[item][item] == 0) SwapRows(item); //Проверка на ноль
for(j=n; j>=item; j--) {
Source[item][j] /= Source[item][item]; //Делим строку i на а[i][i]
}
for(i=item+1; i<n; i++) { //Выполнение операций со строками
for(j=n;j>=item;j--) {
Source[i][j] -= Source[item][j] * Source[i][item];
}
}
if (item % 2) document.write("<table border='1'>");//Вывод итеррации
else    document.write("<table>");
for(i=0; i<n; ++i) {
document.write("<tr>");
for(j=0; j <= n; ++j) {
document.write("<td>" + Source[i][j] + "</td>");
}
document.write("</tr>");
}
document.write("</table>");
}
};
function SwapRows(item) { //Функция перемены строк
for(i=item+1;i<n;i++) {
if(Source[i][item] !== 0) {
for(j=0;j<=n;j++) {
k = Source[i-1][j];
Source[i-1][j] = Source[i][j];
Source[i][j] = k;
}
} 
}
};
function Search_Results() { //Функция поиска и вывода корней
Results[n-1] = Source[n-1][n]/Source[n-1][n-1];
for(i=n-2;i>=0;i--) {
k=0;
for(j=n-1;j>i;j--) {
k = (Source[i][j]*Results[j]) + k;
}
Results[i] = Source[i][n] - k;
}
for(i=n;i>0;i--) {
document.write("<center><h1> x" + i + " = " + Results[i-1] + "</h1></center>");
}
};
