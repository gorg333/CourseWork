<script language="javascript">
NumbOfEquations=prompt("Введите число уравнений системы:"); 
let Source = new Array(NumbOfEquations); //Исходный массива
let Results = new Array(NumbOfEquations); //Массив ответов
let i, j, k;    //Вспомогательные переменные
for(i=0; i<NumbOfEquations; ++i) {
Source[i] = new Array(NumbOfEquations);
Results[i] = new Array(NumbOfEquations);
}
 
for(i=0; i<NumbOfEquations; ++i) { //Заполнение матрицы
for(j=0; j <= NumbOfEquations; ++j) {
Source[i][j] = prompt("Введите значение елемента матрицы:");
}
}
document.write("<table border='3'>"); //Вывод исходной матрицы
for(i=0; i<NumbOfEquations; ++i) {
for(j=0; j <= NumbOfEquations; ++j) {
if(j === 0) document.write("<tr>");
document.write("<td>" + Source[i][j] + "</td>");
if(j === NumbOfEquations) document.write("</tr>");
}
}
document.write("</table>");
Iteration(NumbOfEquations);
Search_Solution();
 
function Iteration(item) { //Функция итеррация
for(item=0;item<(NumbOfEquations-1);item++) { //Цикл выполнения итерраций
if (Source[item][item] == 0) SwapRows(item); //Проверка на ноль
for(j=NumbOfEquations; j>=item; j--) {
Source[item][j] /= Source[item][item]; //Делим строку i на а[i][i]
}
for(i=item+1; i<NumbOfEquations; i++) { //Выполнение операций со строками
for(j=NumbOfEquations;j>=item;j--) {
Source[i][j] -= Source[item][j] * Source[i][item];
}
}
if (item % 2) document.write("<table border='1'>");//Вывод итеррации
else    document.write("<table>");
for(i=0; i<NumbOfEquations; ++i) {
document.write("<tr>");
for(j=0; j <= NumbOfEquations; ++j) {
document.write("<td>" + Source[i][j] + "</td>");
}
document.write("</tr>");
}
document.write("</table>");
}
};
function SwapRows(item) { //Функция перемены строк
for(i=item+1;i<NumbOfEquations;i++) {
if(Source[i][item] !== 0) {
for(j=0;j<=NumbOfEquations;j++) {
k = Source[i-1][j];
Source[i-1][j] = Source[i][j];
Source[i][j] = k;
}
} 
}
};
function Search_Solution() { //Функция поиска и вывода корней
Results[NumbOfEquations-1] = Source[NumbOfEquations-1][NumbOfEquations]/Source[NumbOfEquations-1][NumbOfEquations-1];
for(i=NumbOfEquations-2;i>=0;i--) {
k=0;
for(j=NumbOfEquations-1;j>i;j--) {
k = (Source[i][j]*Results[j]) + k;
}
Results[i] = Source[i][NumbOfEquations] - k;
}
for(i=NumbOfEquations;i>0;i--) {
document.write("<center><h1> x" + i + " = " + Results[i-1] + "</h1></center>");
}
};
</script>
